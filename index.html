RGB To Hex Conversion 
        const checkCode = (hex) => {
            if(hex < 0) hex = 0..toString(16);
            if(hex > 255) hex = 255..toString(16);
            hex = hex.toString(16);
            return hex.length === 1 ? '0' + hex : hex;
        }
        const rgb = (r, g, b) => {
            return [r, g, b].map(checkCode).join('').toUpperCase();
        }
